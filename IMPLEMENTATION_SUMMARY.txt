Show selected plan:

Get today's workout:
  const getTodayWorkout = () => {
    const day = new Date().getDay();
    const idx = day === 0 ? 6 : day - 1;
    return selectedPlan?.workoutSet?.find(w => w.day === idx.toString());
  };

List all plans:
  const { allPlans } = useWorkoutPlans();
  {allPlans.map(plan => <Text key={plan.id}>{plan.label}</Text>)}

Find exercise:
  const exercise = selectedPlan?.workoutSet
    ?.flatMap(w => w.workoutexerciseSet || [])
    ?.find(ex => ex.exercise?.name === 'Bench Press');
